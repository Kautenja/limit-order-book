69  inline void limit_sell(UID order_id, Quantity quantity, Price price) {
70  // put the order into the map
71  orders.emplace(std::piecewise_construct,
72  std::forward_as_tuple(order_id),
73  std::forward_as_tuple(order_id, Side::Sell, quantity, price)
74  );
75  if (buys.best != nullptr && price <= buys.best->key) { // crosses
76  // place a market order with the limit price
77  buys.market(&orders.at(order_id), [&](UID uid) { orders.erase(uid); });
78  if (orders.at(order_id).quantity == 0) { // order filled
79  orders.erase(order_id);
80  return;
81  }
82  }
83  sells.limit(&orders.at(order_id));
84  }

92  inline void limit_buy(UID order_id, Quantity quantity, Price price) {
93  // put the order into the map
94  orders.emplace(std::piecewise_construct,
95  std::forward_as_tuple(order_id),
96  std::forward_as_tuple(order_id, Side::Buy, quantity, price)
97  );
98  if (sells.best != nullptr && price >= sells.best->key) { // crosses
99  // place a market order with the limit price
100  sells.market(&orders.at(order_id), [&](UID uid) { orders.erase(uid); });
101  if (orders.at(order_id).quantity == 0) { // order filled
102  orders.erase(order_id);
103  return;
104  }
105  }
106  buys.limit(&orders.at(order_id));
107  }

116  inline void limit(Side side, UID order_id, Quantity quantity, Price price) {

141  inline void cancel(UID order_id) {
142  auto order = &orders.at(order_id);
143  switch (order->side) { // remove the order from the appropriate side
144  case Side::Sell: { sells.cancel(order); break; }
145  case Side::Buy: { buys.cancel(order); break; }

155  inline void reduce(UID order_id, Quantity quantity) {
156  auto order = &orders.at(order_id);
157  if (quantity > order->quantity) { // trying to remove more than available
158  throw "trying to remove " +
159  std::to_string(quantity) +
160  " from order with " +
161  std::to_string(order->quantity) +
162  " available!";
163  }
164  // remove the quantity from the order and limit
165  order->quantity -= quantity;
166  order->limit->volume -= quantity;
167  switch (order->side) { // remove the order from the appropriate side
168  case Side::Sell: { sells.volume -= quantity; break; }
169  case Side::Buy: { buys.volume -= quantity; break; }
170  }
171  if (order->quantity == 0) { // remove the order if there is no quantity left
172  switch (order->side) { // remove the order from the appropriate side
173  case Side::Sell: { sells.cancel(order); break; }
174  case Side::Buy: { buys.cancel(order); break; }
175  }
176  orders.erase(order_id);
177  }
178  }

249  inline Price price() const {
250  if (sells.best == nullptr && buys.best == nullptr) // empty book
251  return 0;
252  if (sells.best == nullptr) // no sell orders
253  return buys.best->key;
254  if (buys.best == nullptr) // no buy orders
255  return sells.best->key;
256  // return the midpoint
257  return (sells.best->key + buys.best->key) / 2;
258  }

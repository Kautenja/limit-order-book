87 template<>
88 inline void find_best<Side::Buy>(Limit** highest_buy) {
89  if ((*highest_buy)->parent == nullptr) // removing root node
90  // set left because if right existed it would be a better limit already
91  *highest_buy = static_cast<Limit*>((*highest_buy)->left);
92  else // removing non-root node
93  *highest_buy = static_cast<Limit*>((*highest_buy)->parent);
94  if (*highest_buy != nullptr) // highest buy exists
95  *highest_buy = static_cast<Limit*>(BST::max(*highest_buy));
96 }

102 template<>
103 inline void find_best<Side::Sell>(Limit** lowest_sell) {
104  if ((*lowest_sell)->parent == nullptr) // removing root node
105  // set right because if left existed it would be a better limit already
106  *lowest_sell = static_cast<Limit*>((*lowest_sell)->right);
107  else // removing non-root node
108  *lowest_sell = static_cast<Limit*>((*lowest_sell)->parent);
109  if (*lowest_sell != nullptr) // lowest sell exists
110  *lowest_sell = static_cast<Limit*>(BST::min(*lowest_sell));
111 }

133 template<>
134 inline bool can_match<Side::Buy>(Price limit, Price market) {
135  // short circuit on price > 0: 0 is sentinel for no limit
136  // the price should be less than or equal to the best on the buy side.
137  // i.e., the submitted order is a sell limit with a lower price
138  return market == 0 || market <= limit;
139 }
140 
147 template<>
148 inline bool can_match<Side::Sell>(Price limit, Price market) {
149  // short circuit on price > 0: 0 is sentinel for no limit
150  // the price should be greater than or equal to the best on the sell side.
151  // i.e., the submitted order is a buy limit with a higher price
152  return market == 0 || market >= limit;
153 }

172  void clear() {
173  // delete all the limits
174  for(auto item = limits.begin(); item != limits.end(); item++)
175  delete item->second;
176  // clear the pairs from the map
177  limits.clear();
178  // set remaining tracers to 0
179  root = nullptr;
180  best = nullptr;
181  count = 0;
182  volume = 0;
183  }

189  void limit(Order* order) {
190  if (limits.count(order->price) == 0) { // first order at limit price
191  // create the limit and set the order's limit to the limit
192  order->limit = new Limit(order);
193  // the limit price is new, add it to the limit tree
194  BST::insert(
195  reinterpret_cast<BST::Node<Price>**>(&root),
196  static_cast<BST::Node<Price>*>(order->limit)
197  );
198  // set the best price
199  set_best<side>(&best, order->limit);
200  // put the limit into the price to limit mapping
201  limits.emplace(order->limit->key, order->limit);
202  } else { // push to existing queue of orders at limit price
203  // set the order's limit to the limit node for the order's price
204  order->limit = limits.at(order->price);
205  // increment the count of the orders at this limit node
206  ++order->limit->count;
207  // increment the volume of the orders at this limit node
208  order->limit->volume += order->quantity;
209  // add the order to the queue of orders for the limit node
210  DLL::append(
211  reinterpret_cast<DLL::Node**>(&order->limit->order_tail),
212  static_cast<DLL::Node*>(order)
213  );
214  }
215  // update the count and volume for the entire tree
216  ++count;
217  volume += order->quantity;
218  // set the last best price
219  if (best != nullptr) last_best_price = best->key;
220  }
221 
222  // TODO: use object pool for limit reuse?
227  void cancel(Order* order) {
228  // unwrap the limit for the order
229  auto limit_ = order->limit;
230  if (order->prev == nullptr && order->next == nullptr) { // last Order
231  // remove the limit from the tree
232  BST::remove(
233  reinterpret_cast<BST::Node<Price>**>(&root),
234  static_cast<BST::Node<Price>*>(limit_)
235  );
236  // replace the best limit if necessary
237  if (best == limit_)
238  find_best<side>(&best);
239  // remove the limit from the price map
240  limits.erase(limit_->key);
241  delete limit_;
242  } else { // at least 1 other Order in queue
243  --limit_->count;
244  limit_->volume -= order->quantity;
245  // remove the order from the queue (update head and tail)
246  DLL::remove(
247  reinterpret_cast<DLL::Node**>(&limit_->order_head),
248  reinterpret_cast<DLL::Node**>(&limit_->order_tail),
249  static_cast<DLL::Node*>(order)
250  );
251  }
252  // update the count and volume for the entire tree
253  --count;
254  volume -= order->quantity;
255  // set the last best price
256  if (best != nullptr) last_best_price = best->key;
257  }

265  template<typename Callback>
266  void market(Order* order, Callback did_fill) {
267  // find orders until there are none
268  while (best != nullptr && can_match<side>(best->key, order->price)) {
269  // get the next match as the front of the best price
270  auto match = best->order_head;
271  if (match->quantity >= order->quantity) { // current match can fill
272  if (match->quantity == order->quantity) { // limit order filled
273  // remove the current match from the book
274  cancel(match);
275  did_fill(match->uid);
276  } else { // limit order partially filled
277  // remove the market order quantity from the limit quantity
278  match->quantity -= order->quantity;
279  // update the match's limit volume
280  match->limit->volume -= order->quantity;
281  // update the volume for the entire tree
282  volume -= order->quantity;
283  }
284  // clear the remaining quantity for the order
285  order->quantity = 0;
286  return;
287  } // else: current match can NOT fill
288  // decrement the remaining quantity of the market order
289  order->quantity -= match->quantity;
290  // remove the current match from the book
291  cancel(match);
292  did_fill(match->uid);
293  }
294 
295  }
